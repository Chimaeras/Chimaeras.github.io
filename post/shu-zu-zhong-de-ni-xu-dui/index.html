数组中的逆序对

2021-09-15

在数组中的两个数字，如果前面一个数字大于后面的数字，则这两个数字组成一个逆序对。输入一个数组，求出这个数组中的逆序对的总数。

输入: [7,5,6,4]
输出: 5

[7,5] [7,6] [7,4] [5,4] [6,4]

  • 方法1：递归合并

class Solution {
    int[] nums, tmp;

    public int reversePairs ( int[] nums ) {
        this.nums = nums;
        tmp = new int[nums.length];
        return mergeSort( 0, nums.length );
    }

    public int mergeSort ( int left, int right ) {
        // 终止条件：只剩一个节点或超过范围，
        if ( left >= right ) {
            return 0;
        }
        // 取中间数
        int mid = ( left + right ) / 2;
        // 递归划分
        int res = mergeSort( left, mid ) + mergeSort( mid + 1, right );

        // 分为[left,mid]、[mid+1,right]两个区间
        int i = left, j = mid + 1;
        // 辅助数组
        for ( int k = left; k <= right; k++ ) {
            tmp[k] = nums[k];
        }
        // 合并
        for ( int k = left; k <= right; k++ ) {
            // 左子区间已经遍历完
            if ( k == mid + 1 ) {
                nums[k] = tmp[j++];
            }
            // 右子区间已经遍历完 || 左 < 右
            else if ( k == right + 1 || tmp[i] <= tmp[j] ) {
                nums[k] = tmp[i++];
            }
            // 左 > 右
            // 表示左子区间中left后的每一个数都能与右子区间中的right形成一个逆序对，因为区间都是按照大小进行排序的
            else {
                nums[k] = tmp[j++];
                res += mid - left + 1;
            }
        }
        return res;
    }
}
