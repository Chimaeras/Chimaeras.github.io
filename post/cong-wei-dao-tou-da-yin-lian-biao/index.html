从尾到头打印链表

2021-09-08

输入一个链表的头节点，从尾到头反过来返回每个节点的值（用数组返回）。

  • 方法1：反转链表

将链表顺序反转过来，再依次填入数组中

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] reversePrint ( ListNode head ) {
        ListNode pre = null;
        ListNode cur = head;
        ListNode next = null;
        int size = 0;
        while ( cur != null ) {
            next = cur.next;
            cur.next = pre;
            pre = cur;
            cur = next;
            size++;
        }
        int[] res = new int[size];
        int i = 0;
        ListNode newHead = pre;
        while ( newHead != null ) {
            res[i] = newHead.val;
            newHead = newHead.next;
            i++;
        }
        return res;
    }
}

  • 方法2：逆序填充

class Solution {
    public int[] reversePrint ( ListNode head ) {

        int len = 0;
        ListNode cur = head;
        // 获取链表长度
        while ( head != null ) {
            head = head.next;
            len++;
        }
        int[] res = new int[len];

        // 逆序填充进数组中
        while ( cur != null ) {
            res[len - 1] = cur.val;
            len--;
            cur = cur.next;
        }

        return res;
    }
}
