二叉搜索树与双向链表

2021-09-13

输入一棵二叉搜索树，将该二叉搜索树转换成一个排序的循环双向链表。要求不能创建任何新的节点，只能调整树中节点指针的指向。

  • 方法1：中序遍历修改引用

class Solution {
    Node head, pre;
    public Node treeToDoublyList ( Node root ) {
        if ( root == null ) {
            return null;
        }
        // 修改双向引用
        dfs( root );
        // 首尾相连
        pre.right = head;
        head.left = pre;
        return head;
    }

    public void dfs ( Node cur ) {
        // 空节点 直接返回
        if ( cur == null ) {
            return;
        }
        // 递归左子树 ( 左 - 中 - 右 )
        dfs( cur.left );
        // pre为空表示正在访问头节点，所以其前驱节点为空
        if ( pre == null ) {
            // 记为head
            head = cur;
        } else {
            // 前驱节点的右指针指向当前节点
            pre.right = cur;
        }
        // 当前节点的前驱指针指向pre
        cur.left = pre;
        // 往右边挪移
        pre = cur;
        // 递归遍历右节点
        dfs( cur.right );
    }
}

  • 方法2：队列按照中序存储节点，再修改指针

class Solution {
    ArrayList< Node > arrayList = new ArrayList<>();
    public Node treeToDoublyList ( Node root ) {
        if ( root == null ) {
            return null;
        }
        dfs( root );
        int n = arrayList.size();
        Node pre = arrayList.get( n - 1 );
        
        for ( int i = 0; i < n - 1; i++ ) {
            arrayList.get( i ).right = arrayList.get( i + 1 );
            arrayList.get( i + 1 ).left = arrayList.get( i );
        }
        arrayList.get( 0 ).left = arrayList.get( n - 1 );
        arrayList.get( n - 1 ).right = arrayList.get( 0 );
        
        return arrayList.get( 0 );
    }
    public void dfs ( Node root ) {
        if ( root != null ) {
            dfs( root.left );
            arrayList.add( root );
            dfs( root.right );
        }
    }
}
