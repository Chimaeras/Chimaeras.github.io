矩阵中的元素

2021-09-15

给定一个 m x n 二维字符网格 board 和一个字符串单词 word 。如果 word 存在于网格中，返回 true ；否则，返回 false 。

单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。

  • 方法1：回溯遍历

class Solution {
    public boolean exist ( char[][] board, String word ) {
        // 转换成字符数组，方便根据下标取值
        char[] words = word.toCharArray();
        // 以每一个节点为开头尝试一遍
        for ( int i = 0; i < board.length; i++ ) {
            for ( int j = 0; j < board[0].length; j++ ) {
                if ( dfs( i, j, 0, board, words ) ) {
                    return true;
                }
            }
        }
        return false;
    }

    public boolean dfs ( int i, int j, int pos, char[][] board, char[] words ) {
        // 索引越界
        if ( i >= board.length || i < 0 || j >= board[0].length || j < 0 || board[i][j] != words[pos] ) {
            return false;
        }

        // 全部匹配
        if ( pos == words.length - 1 ) {
            return true;
        }

        // 修改当前位置访问标记
        board[i][j] = '\0';
        // 递归四个不同方向
        boolean res = dfs( i + 1, j, pos + 1, board, words )
                || dfs( i - 1, j, pos + 1, board, words )
                || dfs( i, j + 1, pos + 1, board, words )
                || dfs( i, j - 1, pos + 1, board, words );
        // 如果没有匹配成功，回溯之前的修改
        board[i][j] = words[pos];
        return res;
    }
}
