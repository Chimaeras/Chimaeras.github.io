序列化二叉树

2021-09-15

请实现两个函数，分别用来序列化和反序列化二叉树。

你需要设计一个算法来实现二叉树的序列化与反序列化。这里不限定你的序列 / 反序列化算法执行逻辑，你只需要保证一个二叉树可以被序列化为一个字符串并且将这个字符串反序列化为原始的树结构。

  • 方法1：层序遍历，队列维护

class Codec {

    public String serialize ( TreeNode root ) {
        // 节点为空
        if ( root == null ) {
            return "[]";
        }
        // 首尾添加[]
        StringBuilder res = new StringBuilder( "[" );
        // 队列存储节点->层序遍历
        Queue< TreeNode > queue = new LinkedList<>();
        queue.add( root );
        while ( !queue.isEmpty() ) {
            TreeNode node = queue.poll();
            if ( node != null ) {
                // 添加当前节点和逗号，同时将左右子节点放进队列
                res.append( node.val ).append( "," );
                queue.add( node.left );
                queue.add( node.right );
            } else {
                // 表示空节点
                res.append( "null," );
            }
        }
        // 删除最后一个括号
        res.deleteCharAt( res.length() - 1 );
        res.append( "]" );
        return res.toString();
    }

    public TreeNode deserialize ( String data ) {
        // 序列为空，表示头结点为空
        if ( "[]".equals( data ) ) {
            return null;
        }
        // 删除序列化后的首尾中括号,并用逗号分割开
        String[] strings = data.substring( 1, data.length() - 1 ).split( "," );
        TreeNode root = new TreeNode( Integer.parseInt( strings[0] ) );
        Queue< TreeNode > queue = new LinkedList<>();
        queue.add( root );
        // i指向当前头节点的左右子节点
        int i = 1;
        while ( !queue.isEmpty() ) {
            TreeNode node = queue.poll();
            // 左子节点不为空，生成左子节点并放入树中
            if ( !"null".equals( strings[i] ) ) {
                node.left = new TreeNode( Integer.parseInt( strings[i] ) );
                queue.add( node.left );
            }
            i++;
            // 右子节点不为空，生成右子节点并放入树中
            if ( !"null".equals( strings[i] ) ) {
                node.right = new TreeNode( Integer.parseInt( strings[i] ) );
                queue.add( node.right );
            }
            i++;
        }
        return root;
    }
}
